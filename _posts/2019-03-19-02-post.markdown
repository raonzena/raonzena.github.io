# this 5 patterns #

1. Global Reference

~~~javascript
log(this); // this === window
~~~

2. Free Function Invocation

~~~javascript
var fun = function() {
    log(this); 
}
fun(); // this === window
~~~

3. .call or .apply Invocation

~~~javascript
var fun = function() {
    log(this);
}
fun.call(obj); // this === obj
fun.apply(obj); // this === obj
~~~

4. Construction Mode

~~~javascript
var fun = function() {
    log(this);
}
new fun();
~~~

5. Method Invocation

~~~javascript
var obj = {method: fun};
obj.mtd(); // this === obj
~~~
